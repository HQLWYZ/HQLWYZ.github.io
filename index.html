print (" Step1:	Loading in Data...	")

events_array = np.load(import_file_name+'_event_array.npy')# [x, y, charge, roce_charge, reco_adc]

mc_particle_position = np.load(import_file_name+'_mc_array.npy')
#pxq_total = np.load("charge_array_demo.npy")


Q_factor_0 = 0.758
INTER_STRIP = 1
shift_factor = 4


EVENTS_TOTAL = len(events_array)
reconstruction_array= np.zeros(shape = [EVENTS_TOTAL, 2] )
residual_array = np.zeros(shape = [EVENTS_TOTAL, 2] )

pxq = np.array([])
pxq_adc = np.array([])

for iev in range(0, EVENTS_TOTAL):
    
    # shift zero array to array front, for sharing charge distribute reconstruction.
    events_array[iev] = np.vstack((events_array[iev][-shift_factor:],events_array[iev][:-shift_factor]))
    
    if( INTER_STRIP == 0): 
        lk=3
        for ele in range (0, len(events_array[iev][:,2])):
            if events_array[iev][:,2][ele] >0 : 
                pxq = np.append(pxq, events_array[iev][:,2][ele])
                events_array[iev][:,3][ele] = 0
                events_array[iev][:,3][ele] = GAIN*(events_array[iev][:,2][ele]+random.gauss(0,ENC_RMS))
    elif ( INTER_STRIP == 1):
        lk=1
        for ele in range (1, len(events_array[iev][:,2])-1): 
            if(events_array[iev][ele][1]*1000%250 != 0):
                charge_for_sharing = events_array[iev][ele][2]
                tmp = charge_for_sharing*0.5 * Q_factor_0
                events_array[iev][ele-1] = ([0,events_array[iev][ele][1]-0.125,events_array[iev][ele-1][2]+tmp,0,0])
                events_array[iev][ele+1] = ([0,events_array[iev][ele][1]+0.125,events_array[iev][ele+1][2]+tmp,0,0])
                events_array[iev][ele] = 0
                
        for ele in range (0, len(events_array[iev][:,2])):       
            if events_array[iev][:,2][ele] >0 :                
                events_array[iev][:,3][ele] = GAIN*(events_array[iev][:,2][ele]+random.gauss(0,ENC_RMS))

                
                
    for ele in range (0, len(events_array[iev][:,3])):
        if events_array[iev][:,3][ele]< (THRESHOLD+random.gauss(0,THRESHOLD_SMEAR)):
            events_array[iev][:,:][ele] = 0

    for ele in range (0, len(events_array[iev][:,3])):
        if events_array[iev][:,3][ele]>0:
            events_array[iev][:,3][ele] += random.gauss(0,ADC_NOISE)
            events_array[iev][:,4][ele] = min( int((events_array[iev][:,2][ele])/ADC_SLOPE),  (2**ADC_RESOLUTION-1) )
            pxq_adc = np.append(pxq_adc, events_array[iev][:,4][ele])

    # simple CoG find reco position
    if np.sum(events_array[iev][:,4]) > 0 :
        reconstruction_array[iev][0]=  np.sum(events_array[iev][:,4]*events_array[iev][:,0])/np.sum(events_array[iev][:,4])
        reconstruction_array[iev][1]=  np.sum(events_array[iev][:,4]*events_array[iev][:,1])/np.sum(events_array[iev][:,4])
    else:
        reconstruction_array[iev]=([0.,0.])  
        
    # calculate the residual array
    residual_array[iev] = reconstruction_array[iev]- mc_particle_position[iev]

            else:
                cluster_array[iev,1] = 1

            cluster_array[iev,1] = kk[3]/(kk[1]+kk[3])
